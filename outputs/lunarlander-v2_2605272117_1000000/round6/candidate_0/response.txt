    progress_reward = 0.2 * progress

    leg_bonus = 0.1 * (left_leg + right_leg)

    # ---- Terminal bonus – moderate, to avoid dominating the cumulative reward ----
    if terminated:

            abs(x) < 0.3 and abs(angle) < 0.2 and

        )
        terminal_bonus = 20.0 if success else -20.0
        outcome_val = 1.0 if success else -1.0

    # ---- Total reward ----
    total = alive_bonus + progress_reward + leg_bonus + vertical_speed_penalty + terminal_bonus

    # ---- Component dictionary ----
    components = {
        "alive_bonus": alive_bonus,
        "progress_reward": progress_reward,
        "leg_bonus": leg_bonus,
        "vertical_speed_penalty": vertical_speed_penalty,
        "terminal_bonus": terminal_bonus,

    }
