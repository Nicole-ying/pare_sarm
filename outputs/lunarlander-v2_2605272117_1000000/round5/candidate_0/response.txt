import math

def compute_reward(state, m_power, s_power, terminated):
    # Unpack observation
    x = state[0]
    y = state[1]
    vx = state[2]
    vy = state[3]
    angle = state[4]
    left_leg = state[6]
    right_leg = state[7]

    # ----- Per-step shaping components -----
    # 1. Proximity: guide toward proper position and orientation
    x_progress = 1.0 - min(1.0, abs(x) / 1.5)
    y_progress = 1.0 - min(1.0, abs(y) / 1.5)      # encourage descent to pad height
    angle_progress = 1.0 - min(1.0, abs(angle) / math.pi)
    proximity = 0.3 * x_progress + 0.3 * y_progress + 0.4 * angle_progress

    # 2. Speed penalty: discourage high horizontal/vertical velocity
    total_speed = abs(vx) + abs(vy)
    speed_penalty = -0.3 * min(5.0, total_speed)

    # 3. Leg contact bonus: small reward for stable ground contact (reduced to avoid farming)
    leg_bonus = 0.1 * (left_leg + right_leg)

    # 4. Fuel penalty: small cost for firing engines
    fuel_penalty = -0.1 * (m_power + abs(s_power))

    # 5. Terminal bonus: large reward/penalty at episode end
    if terminated:
        # Success: both legs on ground, near pad, upright, nearly zero velocity
        success = (
            left_leg == 1.0 and right_leg == 1.0 and
            abs(x) < 0.3 and abs(angle) < 0.2 and
            abs(vx) < 0.5 and abs(vy) < 0.5
        )
        outcome_val = 1.0 if success else -1.0
        terminal_bonus = 100.0 if success else -100.0
    else:
        outcome_val = 0.0
        terminal_bonus = 0.0

    # Total per-step reward (includes terminal if terminated)
    total = proximity + speed_penalty + leg_bonus + fuel_penalty + terminal_bonus

    # Component dictionary
    components = {
        "proximity": proximity,
        "speed_penalty": speed_penalty,
        "leg_bonus": leg_bonus,
        "fuel_penalty": fuel_penalty,
        "terminal_bonus": terminal_bonus,
        "_outcome": outcome_val
    }

    return float(total), components
