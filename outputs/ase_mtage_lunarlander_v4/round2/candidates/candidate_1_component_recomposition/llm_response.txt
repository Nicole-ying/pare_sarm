import math


def _safe_float(x, default=0.0):
    try:
        value = float(x)
    except (ValueError, TypeError):
        return default
    if not math.isfinite(value):
        return default
    return value


def compute_reward(obs, action, next_obs, terminated, truncated, info):
    # Extract current and next state features from the observation array.
    x0 = _safe_float(obs[0]) if len(obs) > 0 else 0.0
    y0 = _safe_float(obs[1]) if len(obs) > 1 else 0.0
    x1 = _safe_float(next_obs[0]) if len(next_obs) > 0 else 0.0
    y1 = _safe_float(next_obs[1]) if len(next_obs) > 1 else 0.0
    vx1 = _safe_float(next_obs[2]) if len(next_obs) > 2 else 0.0
    vy1 = _safe_float(next_obs[3]) if len(next_obs) > 3 else 0.0
    angle1 = _safe_float(next_obs[4]) if len(next_obs) > 4 else 0.0
    angvel1 = _safe_float(next_obs[5]) if len(next_obs) > 5 else 0.0
    left_leg1 = _safe_float(next_obs[6]) if len(next_obs) > 6 else 0.0
    right_leg1 = _safe_float(next_obs[7]) if len(next_obs) > 7 else 0.0

    # Distance to the landing pad (origin).
    prev_dist = math.sqrt(x0 * x0 + y0 * y0)
    next_dist = math.sqrt(x1 * x1 + y1 * y1)
    progress_delta = prev_dist - next_dist  # positive = moving closer

    # Stage definition: near when distance < 0.35.
    near_threshold = 0.35
    far_stage = 1.0 if next_dist >= near_threshold else 0.0
    near_stage = 1.0 - far_stage

    # ---- Progress term (far stage only) ----
    # Encourage closing distance to the landing pad.
    approach_progress = far_stage * progress_delta

    # ---- Stability/safety term (near stage only) ----
    # Reward good behavior (both legs contact, low speeds, small angle)
    # rather than heavily penalizing. This avoids counterproductive over-penalization.
    leg_bonus = (left_leg1 + right_leg1) * 2.0  # 0..4
    speed_penalty = (abs(vx1) + abs(vy1)) * 0.5
    angle_penalty = abs(angle1) * 2.0
    angvel_penalty = abs(angvel1) * 0.2
    safety_reward = leg_bonus - speed_penalty - angle_penalty - angvel_penalty
    # Clamp to avoid extreme negative values dominating during early near-stage entries
    safety_reward = max(-2.0, safety_reward)
    safety_compliance = near_stage * safety_reward

    # ---- Terminal handling ----
    terminal_crash_penalty = 0.0
    terminal_safe_bonus = 0.0
    if terminated:
        both_legs = (left_leg1 > 0.5 and right_leg1 > 0.5)
        # Crash heuristics: high vertical speed, large tilt, high horizontal speed, or no leg contact.
        crash = (abs(vy1) > 0.8 or abs(angle1) > 0.5 or abs(vx1) > 0.5 or not both_legs)
        # Safe landing heuristics: both legs contact, low speeds, small angle.
        safe = (both_legs and abs(vy1) < 0.2 and abs(vx1) < 0.2 and abs(angle1) < 0.2)
        if crash:
            terminal_crash_penalty = -10.0
        elif safe:
            terminal_safe_bonus = 20.0
        # else ambiguous termination – no bonus or penalty.

    # ---- Timeout penalty for low progress ----
    low_progress_timeout = -1.0 if truncated and next_dist >= near_threshold else 0.0

    components = {
        "approach_progress": approach_progress,
        "safety_compliance": safety_compliance,
        "terminal_crash_penalty": terminal_crash_penalty,
        "terminal_safe_bonus": terminal_safe_bonus,
        "low_progress_timeout": low_progress_timeout,
    }
    total_reward = (
        4.0 * approach_progress
        + 1.5 * safety_compliance
        + terminal_crash_penalty
        + terminal_safe_bonus
        + low_progress_timeout
    )
    return float(total_reward), components
